    ```python
    """CMP (Custo Médio Ponderado) — Preço Médio Ponderado calculation engine.

    B3/CVM-mandated cost basis methodology for Brazilian investment accounts.
    All arithmetic uses Python Decimal (18 decimal places) — never float.

    References:
    - B3: Instrução CVM 8/1979 (custo médio ponderado)
    - CVM Resolution 19/2021 (IR calculation on equity gains)

    Pure functions: no DB, no Redis, no FastAPI coupling.
    The service layer (portfolio/service.py) loads data from DB and calls these functions.
    """
    from __future__ import annotations

    from dataclasses import dataclass, field
    from datetime import date
    from decimal import Decimal, ROUND_HALF_UP
    from typing import Any

    # 8 decimal places — sufficient for CMP precision without float drift
    _DP = Decimal("0.00000001")
    _ZERO = Decimal("0")


    @dataclass
    class Position:
        """Current position state for a single ticker.

        Immutable by convention — all functions return new Position instances.
        Mutation causes hard-to-debug CMP drift in multi-transaction sequences.
        """
        ticker: str
        quantity: Decimal
        cmp: Decimal          # Custo Médio Ponderado
        total_cost: Decimal   # quantity × cmp (invariant across corporate events)
        asset_class: str      # AssetClass enum value


    def _quantize(value: Decimal) -> Decimal:
        """Round to 8 decimal places with ROUND_HALF_UP."""
        return value.quantize(_DP, rounding=ROUND_HALF_UP)


    def apply_buy(position: Position, qty: Decimal, unit_price: Decimal,
                  brokerage_fee: Decimal | None = None) -> Position:
        """Apply a buy transaction — recalculates CMP.

        B3 formula: new_cmp = (qty_held × cmp_prev + qty_bought × effective_price) / total_qty
        Brokerage fee is included in the effective price (B3 cost basis rule).
        """
        if qty <= _ZERO:
            raise ValueError(f"Buy quantity must be positive, got {qty}")

        # Effective price includes brokerage fee proportionally
        fee = brokerage_fee or _ZERO
        effective_price = _quantize((qty * unit_price + fee) / qty)

        if position.quantity == _ZERO:
            new_cmp = effective_price
        else:
            new_cmp = _quantize(
                (position.quantity * position.cmp + qty * effective_price) /
                (position.quantity + qty)
            )

        new_qty = position.quantity + qty
        return Position(
            ticker=position.ticker,
            quantity=new_qty,
            cmp=new_cmp,
            total_cost=_quantize(new_qty * new_cmp),
            asset_class=position.asset_class,
        )


    def apply_sell(position: Position, qty: Decimal, sale_price: Decimal) -> tuple[Position, Decimal]:
        """Apply a sell transaction — CMP is UNCHANGED. Returns (new_position, realized_pnl).

        P&L = (sale_price - cmp) × qty
        Partial sell: CMP unchanged, quantity reduced.
        """
        if qty <= _ZERO:
            raise ValueError(f"Sell quantity must be positive, got {qty}")
        if qty > position.quantity:
            raise ValueError(
                f"Cannot sell {qty} shares — only {position.quantity} held for {position.ticker}"
            )

        realized_pnl = _quantize((sale_price - position.cmp) * qty)
        new_qty = position.quantity - qty

        return (
            Position(
                ticker=position.ticker,
                quantity=new_qty,
                cmp=position.cmp,  # CMP NEVER changes on sell
                total_cost=_quantize(new_qty * position.cmp),
                asset_class=position.asset_class,
            ),
            realized_pnl,
        )


    def apply_corporate_event(
        position: Position,
        action_type: str,  # CorporateActionType value
        factor: Decimal,
        issue_price: Decimal | None = None,  # bonificacao only
    ) -> Position:
        """Apply a corporate event — adjusts quantity and CMP, total_cost is invariant.

        desdobramento (split): qty × factor, cmp ÷ factor
        grupamento (reverse split): qty ÷ factor, cmp × factor
        bonificacao (bonus shares): qty += qty × factor; new_cmp = weighted average with issue_price
        """
        if action_type == "desdobramento":
            new_qty = _quantize(position.quantity * factor)
            new_cmp = _quantize(position.cmp / factor)
            return Position(
                ticker=position.ticker,
                quantity=new_qty,
                cmp=new_cmp,
                total_cost=position.total_cost,  # invariant
                asset_class=position.asset_class,
            )

        elif action_type == "grupamento":
            new_qty = _quantize(position.quantity / factor)
            new_cmp = _quantize(position.cmp * factor)
            return Position(
                ticker=position.ticker,
                quantity=new_qty,
                cmp=new_cmp,
                total_cost=position.total_cost,  # invariant
                asset_class=position.asset_class,
            )

        elif action_type == "bonificacao":
            if issue_price is None:
                raise ValueError("bonificacao requires issue_price")
            bonus_qty = _quantize(position.quantity * factor)
            new_qty = position.quantity + bonus_qty
            new_cmp = _quantize(
                (position.quantity * position.cmp + bonus_qty * issue_price) / new_qty
            )
            return Position(
                ticker=position.ticker,
                quantity=new_qty,
                cmp=new_cmp,
                total_cost=_quantize(new_qty * new_cmp),
                asset_class=position.asset_class,
            )

        else:
            raise ValueError(f"Unknown corporate action type: {action_type}")


    def build_position_from_history(
        ticker: str,
        asset_class: str,
        transactions: list[Any],  # list of Transaction-like objects
        corporate_actions: list[Any],  # list of CorporateAction-like objects
    ) -> Position:
        """Build current Position by replaying all transactions and corporate events in date order.

        B3 rule: on the same date, corporate actions (ex-date) are applied BEFORE transactions.
        This ensures splits/bonuses adjust the cost basis before any same-day trades.
        """
        position = Position(
            ticker=ticker,
            quantity=_ZERO,
            cmp=_ZERO,
            total_cost=_ZERO,
            asset_class=asset_class,
        )

        # Build event timeline: (date, priority, event)
        # priority 0 = corporate action (applied first), 1 = transaction
        timeline: list[tuple[date, int, Any]] = []

        for ca in corporate_actions:
            timeline.append((ca.action_date, 0, ca))
        for tx in transactions:
            timeline.append((tx.transaction_date, 1, tx))

        timeline.sort(key=lambda e: (e[0], e[1]))

        for _, _, event in timeline:
            if hasattr(event, "action_type"):
                # CorporateAction
                issue_price = getattr(event, "issue_price", None)
                position = apply_corporate_event(
                    position, event.action_type, event.factor, issue_price
                )
            elif event.transaction_type in ("buy",):
                position = apply_buy(
                    position,
                    event.quantity,
                    event.unit_price,
                    getattr(event, "brokerage_fee", None),
                )
            elif event.transaction_type in ("sell",):
                position, _ = apply_sell(position, event.quantity, event.unit_price)
            # dividend, jscp, amortization: do not affect CMP or quantity

        return position
    ```

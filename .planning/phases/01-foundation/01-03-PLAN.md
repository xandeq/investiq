    **Migration 0002 (backend/alembic/versions/0002_add_rls_policies.py):**

    This migration MUST be written manually — Alembic autogenerate does not detect RLS policies (known pitfall from RESEARCH.md).

    ```python
    def upgrade() -> None:
        # Create application role — NOT superuser, so RLS applies to it
        op.execute("CREATE ROLE app_user LOGIN PASSWORD 'change_in_production'")
        op.execute("GRANT CONNECT ON DATABASE investiq TO app_user")
        op.execute("GRANT USAGE ON SCHEMA public TO app_user")
        op.execute("GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO app_user")
        op.execute("GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA public TO app_user")

        # Enable RLS on users table
        # FORCE ROW LEVEL SECURITY applies even to the table owner — critical!
        op.execute("ALTER TABLE users ENABLE ROW LEVEL SECURITY")
        op.execute("ALTER TABLE users FORCE ROW LEVEL SECURITY")

        # Isolation policy — uses current_setting() directly (LEAKPROOF function)
        # Do NOT wrap in a custom function — prevents PostgreSQL index use
        # NULLIF handles the case where rls.tenant_id is not set (returns empty string → NULL)
        op.execute("""
            CREATE POLICY tenant_isolation ON users
              AS PERMISSIVE
              FOR ALL
              TO app_user
              USING (tenant_id = NULLIF(current_setting('rls.tenant_id', TRUE), '')::uuid)
              WITH CHECK (tenant_id = NULLIF(current_setting('rls.tenant_id', TRUE), '')::uuid)
        """)

        # Also enable on refresh_tokens and verification_tokens tables
        op.execute("ALTER TABLE refresh_tokens ENABLE ROW LEVEL SECURITY")
        op.execute("ALTER TABLE refresh_tokens FORCE ROW LEVEL SECURITY")
        op.execute("""
            CREATE POLICY tenant_isolation ON refresh_tokens
              AS PERMISSIVE FOR ALL TO app_user
              USING (user_id IN (
                SELECT id FROM users
                WHERE tenant_id = NULLIF(current_setting('rls.tenant_id', TRUE), '')::uuid
              ))
        """)

    def downgrade() -> None:
        op.execute("DROP POLICY IF EXISTS tenant_isolation ON refresh_tokens")
        op.execute("ALTER TABLE refresh_tokens DISABLE ROW LEVEL SECURITY")
        op.execute("DROP POLICY IF EXISTS tenant_isolation ON users")
        op.execute("ALTER TABLE users DISABLE ROW LEVEL SECURITY")
        op.execute("REVOKE ALL ON ALL TABLES IN SCHEMA public FROM app_user")
        op.execute("DROP ROLE IF EXISTS app_user")
    ```

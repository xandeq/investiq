```python
from app.modules.advisor.schemas import EntrySignal
from app.modules.swing_trade.service import calculate_rsi_ma
from app.modules.opportunity_detector.agents.recommendation import generate_recommendation
import redis as redis_lib

async def get_portfolio_entry_signals(
    tenant_db: AsyncSession,
    global_db: AsyncSession,
    tenant_id: str,
) -> list[EntrySignal]:
    """
    On-demand signals for user's owned assets.
    Cached <5min to balance freshness with cost.
    """
    cache_key = f"entry_signals:portfolio:{tenant_id}"
    
    # Check cache
    try:
        r = redis_lib.Redis.from_url(os.environ.get("REDIS_URL", "redis://localhost:6379/0"), decode_responses=True)
        cached = r.get(cache_key)
        if cached:
            import json
            return [EntrySignal(**item) for item in json.loads(cached)]
    except Exception as exc:
        logger.warning("get_portfolio_entry_signals: Cache read failed: %s", exc)
    
    # Fetch portfolio positions
    tx_result = await tenant_db.execute(
        select(Transaction.ticker).where(
            Transaction.tenant_id == tenant_id,
            Transaction.transaction_type == "buy",
            Transaction.deleted_at.is_(None),
        ).distinct()
    )
    tickers = [row[0] for row in tx_result.all()]
    
    if not tickers:
        return []
    
    # Calculate signals for each ticker
    health = await compute_portfolio_health(tenant_db, global_db, tenant_id)
    signals = []
    
    for ticker in tickers[:10]:  # Limit to top 10 to avoid API explosion
        try:
            technical = await calculate_rsi_ma(ticker)
            recommendation = await generate_recommendation(
                ticker=ticker,
                health_context={
                    "health_score": health.health_score,
                    "biggest_risk": health.biggest_risk,
                },
                technical_signals=technical,
            )
            
            signals.append(EntrySignal(
                ticker=ticker,
                suggested_amount_brl=str(recommendation["suggested_amount_brl"]),
                target_upside_pct=recommendation["target_upside_pct"],
                timeframe_days=recommendation["timeframe_days"],
                stop_loss_pct=recommendation["stop_loss_pct"],
                rsi=technical.get("rsi"),
                ma_signal=technical.get("signal"),
                generated_at=datetime.utcnow(),
            ))
        except Exception as exc:
            logger.warning(f"get_portfolio_entry_signals: Failed for {ticker}: {exc}")
            continue
    
    # Cache result
    try:
        r = redis_lib.Redis.from_url(os.environ.get("REDIS_URL", "redis://localhost:6379/0"), decode_responses=True)
        import json
        data = [s.model_dump(mode="json") for s in signals]
        r.setex(cache_key, 300, json.dumps(data))  # 5 min cache
    except Exception as exc:
        logger.error("get_portfolio_entry_signals: Cache write failed: %s", exc)
    
    return signals


async def get_universe_entry_signals(
    global_db: AsyncSession,
) -> list[EntrySignal]:
    """
    Batch signals for universe (from overnight Celery job).
    Reads from Redis cache (no DB query).
    """
    cache_key = "entry_signals:universe"
    
    try:
        r = redis_lib.Redis.from_url(os.environ.get("REDIS_URL", "redis://localhost:6379/0"), decode_responses=True)
        cached = r.get(cache_key)
        if cached:
            import json
            return [EntrySignal(**item) for item in json.loads(cached)]
    except Exception as exc:
        logger.warning("get_universe_entry_signals: Cache read failed: %s", exc)
    
    return []  # Empty if batch hasn't run yet
```

```python
from celery import shared_task
from app.core.db import get_global_db_session
from app.modules.advisor.service import get_universe_entry_signals
import json
import redis as redis_lib
import os

@shared_task(name="advisor.refresh_universe_entry_signals")
def refresh_universe_entry_signals_batch():
    """
    Nightly Celery beat task: calculate entry signals for ~100 top universe assets,
    store in Redis cache key: entry_signals:universe
    """
    try:
        # Get async session
        import asyncio
        from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession
        from sqlalchemy.orm import sessionmaker
        
        db_url = os.environ.get("DATABASE_URL", "...")
        engine = create_async_engine(db_url, echo=False)
        AsyncSessionLocal = sessionmaker(engine, class_=AsyncSession, expire_on_commit=False)
        
        async def _run():
            async with AsyncSessionLocal() as session:
                # Get top 100 tickers by market cap
                from app.modules.market_universe.models import ScreenerSnapshot
                from sqlalchemy import select, func
                
                latest_date = await session.execute(
                    select(func.max(ScreenerSnapshot.snapshot_date))
                )
                date_val = latest_date.scalar()
                
                top_tickers = await session.execute(
                    select(ScreenerSnapshot.ticker)
                    .where(ScreenerSnapshot.snapshot_date == date_val)
                    .order_by(ScreenerSnapshot.market_cap.desc())
                    .limit(100)
                )
                tickers = [row[0] for row in top_tickers.all()]
                
                signals = []
                for ticker in tickers:
                    try:
                        from app.modules.swing_trade.service import calculate_rsi_ma
                        from app.modules.opportunity_detector.agents.recommendation import generate_recommendation
                        
                        technical = await calculate_rsi_ma(ticker)
                        recommendation = await generate_recommendation(
                            ticker=ticker,
                            health_context={},
                            technical_signals=technical,
                        )
                        
                        from app.modules.advisor.schemas import EntrySignal
                        from datetime import datetime
                        
                        signal = EntrySignal(
                            ticker=ticker,
                            suggested_amount_brl=str(recommendation["suggested_amount_brl"]),
                            target_upside_pct=recommendation["target_upside_pct"],
                            timeframe_days=recommendation["timeframe_days"],
                            stop_loss_pct=recommendation["stop_loss_pct"],
                            rsi=technical.get("rsi"),
                            ma_signal=technical.get("signal"),
                            generated_at=datetime.utcnow(),
                        )
                        signals.append(signal.model_dump(mode="json"))
                    except Exception as exc:
                        logger.warning(f"refresh_universe_entry_signals: Failed for {ticker}: {exc}")
                        continue
                
                # Store in Redis
                r = redis_lib.Redis.from_url(os.environ.get("REDIS_URL", "redis://localhost:6379/0"), decode_responses=True)
                r.setex("entry_signals:universe", 86400, json.dumps(signals))  # 24 hour cache
                
                return len(signals)
        
        return asyncio.run(_run())
    
    except Exception as exc:
        logger.error(f"refresh_universe_entry_signals_batch failed: {exc}")
```

```typescript
import { useQuery } from "@tanstack/react-query";

export interface EntrySignal {
  ticker: string;
  suggested_amount_brl: string;
  target_upside_pct: number;
  timeframe_days: number;
  stop_loss_pct: number;
  rsi: number | null;
  ma_signal: string | null;
  generated_at: string;
}

export function usePortfolioEntrySignals(enabled?: boolean) {
  return useQuery({
    queryKey: ["entry-signals-portfolio"],
    queryFn: async () => {
      const res = await fetch("/api/advisor/signals/portfolio");
      if (!res.ok) throw new Error(`Failed to fetch: ${res.status}`);
      return res.json() as Promise<EntrySignal[]>;
    },
    enabled: enabled !== false,
    refetchInterval: 4 * 60 * 1000,  // 4 minute refetch (cache is 5 min)
  });
}

export function useUniverseEntrySignals(enabled?: boolean) {
  return useQuery({
    queryKey: ["entry-signals-universe"],
    queryFn: async () => {
      const res = await fetch("/api/advisor/signals/universe");
      if (!res.ok) throw new Error(`Failed to fetch: ${res.status}`);
      return res.json() as Promise<EntrySignal[]>;
    },
    enabled: enabled !== false,
    refetchInterval: 60 * 60 * 1000,  // 1 hour refetch (batch is 24 hours)
  });
}
```

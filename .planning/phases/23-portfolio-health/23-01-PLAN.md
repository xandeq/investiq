```python
@router.get(
    "/health",
    response_model=PortfolioHealth,
    summary="Portfolio health check (cached)",
    tags=["advisor"],
)
@limiter.limit("30/minute")
async def get_portfolio_health(
    request: Request,
    current_user: dict = Depends(get_current_user),
    tenant_db: AsyncSession = Depends(get_tenant_db),
    global_db: AsyncSession = Depends(get_global_db),
) -> PortfolioHealth:
    """Return cached portfolio health (score, biggest_risk, passive_income, underperformers).
    
    Cache TTL: 3600 seconds (1 hour).
    Invalidates on: POST /advisor/health/refresh, transaction import, position add.
    """
    tenant_id = current_user["tenant_id"]
    cache_key = f"portfolio_health:{tenant_id}"
    
    try:
        r = _get_redis()
        cached = r.get(cache_key)
        if cached:
            # Deserialize from JSON (Redis returns string, FastAPI will validate)
            import json
            data = json.loads(cached)
            return PortfolioHealth(**data)
    except Exception as exc:
        logger.warning("get_portfolio_health: Redis error (will compute fresh): %s", exc)
    
    # Compute fresh
    health = await compute_portfolio_health(tenant_db, global_db, tenant_id)
    
    # Cache result
    try:
        import json
        r = _get_redis()
        data_dict = health.model_dump(mode="json")
        r.setex(cache_key, 3600, json.dumps(data_dict))
    except Exception as exc:
        logger.error("get_portfolio_health: Failed to cache: %s", exc)
    
    return health


@router.post(
    "/health/refresh",
    response_model=PortfolioHealth,
    summary="Refresh portfolio health (bypass cache)",
    tags=["advisor"],
)
@limiter.limit("10/minute")
async def refresh_portfolio_health(
    request: Request,
    current_user: dict = Depends(get_current_user),
    tenant_db: AsyncSession = Depends(get_tenant_db),
    global_db: AsyncSession = Depends(get_global_db),
) -> PortfolioHealth:
    """Compute fresh portfolio health, invalidate cache, return result."""
    tenant_id = current_user["tenant_id"]
    cache_key = f"portfolio_health:{tenant_id}"
    
    # Clear cache
    try:
        r = _get_redis()
        r.delete(cache_key)
    except Exception as exc:
        logger.warning("refresh_portfolio_health: Failed to clear cache: %s", exc)
    
    # Compute fresh
    health = await compute_portfolio_health(tenant_db, global_db, tenant_id)
    
    # Re-cache
    try:
        import json
        r = _get_redis()
        data_dict = health.model_dump(mode="json")
        r.setex(cache_key, 3600, json.dumps(data_dict))
    except Exception as exc:
        logger.error("refresh_portfolio_health: Failed to cache: %s", exc)
    
    return health
```

```typescript
import { useQuery } from "@tanstack/react-query";

export interface PortfolioHealthResponse {
  health_score: number;
  biggest_risk: string | null;
  passive_income_monthly_brl: string; // Decimal as string
  underperformers: string[];
  data_as_of: string | null; // ISO datetime
  total_assets: number;
  has_portfolio: boolean;
}

export interface UsePortfolioHealthOptions {
  /** If true, skip fetching (useful when portfolio is empty) */
  enabled?: boolean;
  /** Refetch interval in ms (default: no auto-refetch) */
  refetchInterval?: number;
}

export function usePortfolioHealth(options?: UsePortfolioHealthOptions) {
  return useQuery({
    queryKey: ["portfolio-health"],
    queryFn: async () => {
      const res = await fetch("/api/advisor/health");
      if (!res.ok) {
        throw new Error(`Failed to fetch portfolio health: ${res.status}`);
      }
      return res.json() as Promise<PortfolioHealthResponse>;
    },
    enabled: options?.enabled !== false,
    refetchInterval: options?.refetchInterval,
    staleTime: 5 * 60 * 1000, // 5 minutes (server caches 1 hour, but client can cache 5 min)
  });
}

export function useRefreshPortfolioHealth() {
  return async () => {
    const res = await fetch("/api/advisor/health/refresh", { method: "POST" });
    if (!res.ok) {
      throw new Error(`Failed to refresh portfolio health: ${res.status}`);
    }
    return res.json() as Promise<PortfolioHealthResponse>;
  };
}
```

interface HealthCheckCardProps {

function HealthCheckCard({ health, isLoading, onRefresh }: HealthCheckCardProps) {
  const [isRefreshing, setIsRefreshing] = React.useState(false);
  
  const handleRefresh = async () => {
    setIsRefreshing(true);
    try {
      await onRefresh();
    } finally {
      setIsRefreshing(false);
    }
  };
  
  // Color coding for health score
  const scoreColor = health.health_score >= 80 ? "text-green-600" 
                   : health.health_score >= 60 ? "text-yellow-600"
                   : "text-red-600";
  const bgColor = health.health_score >= 80 ? "bg-green-50"
                : health.health_score >= 60 ? "bg-yellow-50"
                : "bg-red-50";
  
  return (
    <div className={`${bgColor} border rounded-lg p-6 mb-8`}>
      <div className="flex justify-between items-start">
        <div>
          <h2 className="text-lg font-semibold mb-4">Health Check</h2>
          
          <div className="mb-6">
            <div className={`text-4xl font-bold ${scoreColor}`}>
              {health.health_score}/100
            </div>
            <p className="text-sm text-muted-foreground mt-1">
              {health.health_score >= 80 ? "Portfólio equilibrado"
             : health.health_score >= 60 ? "Atenção — pontos de melhoria"
             : "Revisar — concentração ou risco elevado"}
            </p>
          </div>
          
          {health.biggest_risk && (
            <div className="mb-4 p-3 bg-orange-100 border border-orange-300 rounded text-sm">
              <strong>Maior Risco:</strong> {health.biggest_risk}
            </div>
          )}
          
          <div className="grid grid-cols-2 gap-4 mb-4">
            <div>
              <p className="text-xs text-muted-foreground">Renda Passiva Mensal</p>

            <div>
              <p className="text-xs text-muted-foreground">Ativos Distintos</p>

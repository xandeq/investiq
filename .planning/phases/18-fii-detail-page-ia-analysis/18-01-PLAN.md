    ```python
    """BRAPI FII data fetcher with Redis caching (Phase 18).
    
    Fetches FII-specific data: current price, DY history (monthly), P/VP,
    portfolio fields from summaryProfile. Same caching pattern as data.py.
    """
    from __future__ import annotations
    import json, logging, os
    from datetime import datetime, timedelta
    from collections import defaultdict
    import requests
    from app.modules.analysis.data import (
        _BRAPI_BASE_URL, _CACHE_TTL_SECONDS, _resolve_brapi_token,
        _get_sync_redis, _extract, DataFetchError,
    )

    logger = logging.getLogger(__name__)

    def fetch_fii_data(ticker: str) -> dict:
        """Fetch FII detail data from BRAPI with 24h Redis cache."""
        ticker_upper = ticker.upper()
        cache_key = f"brapi:fii_detail:{ticker_upper}"
        
        # Redis cache check
        try:
            r = _get_sync_redis()
            cached = r.get(cache_key)
            if cached:
                return json.loads(cached)
        except Exception as exc:
            logger.warning("Redis cache miss for FII %s: %s", ticker_upper, exc)
        
        # Fetch from BRAPI
        token = _resolve_brapi_token()
        params = {"modules": "dividendsData,summaryProfile", "token": token} if token else {"modules": "dividendsData,summaryProfile"}
        
        try:
            url = f"{_BRAPI_BASE_URL}/quote/{ticker_upper}"
            resp = requests.get(url, params=params, timeout=15)
            
            # Handle MODULES_NOT_AVAILABLE (400 from BRAPI)
            if resp.status_code == 400:
                try:
                    err_body = resp.json()
                    if err_body.get("code") == "MODULES_NOT_AVAILABLE" or "MODULES_NOT_AVAILABLE" in str(err_body):
                        # Fallback: fetch base quote only
                        resp2 = requests.get(f"{_BRAPI_BASE_URL}/quote/{ticker_upper}", params={"token": token} if token else {}, timeout=15)
                        resp2.raise_for_status()
                        data = resp2.json()
                    else:
                        raise DataFetchError(ticker_upper, "BRAPI", f"HTTP 400: {err_body}")
                except (json.JSONDecodeError, ValueError):
                    raise DataFetchError(ticker_upper, "BRAPI", f"HTTP {resp.status_code}")
            else:
                resp.raise_for_status()
                data = resp.json()
        except DataFetchError:
            raise
        except Exception as exc:
            raise DataFetchError(ticker_upper, "BRAPI", str(exc))
        
        results = data.get("results", [])
        if not results:
            raise DataFetchError(ticker_upper, "BRAPI", "No results returned")
        
        r_data = results[0]
        current_price = r_data.get("regularMarketPrice")
        daily_volume = r_data.get("regularMarketVolume")
        key_stats = r_data.get("defaultKeyStatistics", {})
        summary_profile = r_data.get("summaryProfile", {})
        dividends_data = r_data.get("dividendsData", {})
        
        book_value = _extract(key_stats, "bookValue")
        price_to_book = _extract(key_stats, "priceToBook")
        
        # Parse cashDividends into monthly aggregation (last 12 months)
        cash_dividends = dividends_data.get("cashDividends", []) if dividends_data else []
        dividends_monthly, dy_12m, last_dividend = _parse_dividends_monthly(cash_dividends, current_price)
        
        # Portfolio fields from summaryProfile (defensive — may be None)
        portfolio = {
            "num_imoveis": summary_profile.get("numberOfProperties") or summary_profile.get("numImoveis"),
            "tipo_contrato": summary_profile.get("contractType") or summary_profile.get("tipoContrato"),
            "vacancia": summary_profile.get("vacancy") or summary_profile.get("vacancia"),
        }
        
        # P/VP: use priceToBook from key_stats, or compute if both available
        pvp = price_to_book
        if pvp is None and current_price and book_value and book_value > 0:
            pvp = round(current_price / book_value, 2)
        
        result = {
            "current_price": current_price,
            "pvp": pvp,
            "dy_12m": dy_12m,
            "dividends_monthly": dividends_monthly,
            "portfolio": portfolio,
            "last_dividend": last_dividend,
            "daily_liquidity": daily_volume,
            "book_value": book_value,
        }
        
        # Cache in Redis (24h TTL)
        try:
            r = _get_sync_redis()
            r.setex(cache_key, _CACHE_TTL_SECONDS, json.dumps(result))
        except Exception as exc:
            logger.warning("Redis cache write failed for FII %s: %s", ticker_upper, exc)
        
        return result


    def _parse_dividends_monthly(cash_dividends: list, current_price: float | None) -> tuple[list, float | None, float | None]:
        """Parse BRAPI cashDividends into monthly aggregation for last 12 months."""
        if not cash_dividends:
            return [], None, None
        
        now = datetime.now()
        twelve_months_ago = now - timedelta(days=365)
        
        # Aggregate by month
        monthly = defaultdict(float)
        last_div = None
        
        for div in cash_dividends:
            rate = div.get("rate")
            if rate is None:
                continue
            
            # Parse date (handle multiple formats)
            date_str = div.get("paymentDate", "")
            parsed_date = _parse_date(date_str)
            if parsed_date is None:
                continue
            
            if last_div is None or parsed_date > _parse_date(cash_dividends[0].get("paymentDate", "")) if False else True:
                pass  # last_div handled below
            
            if parsed_date >= twelve_months_ago:
                month_key = parsed_date.strftime("%Y-%m")
                monthly[month_key] += rate
        
        # Sort by month, take last 12
        sorted_months = sorted(monthly.items())[-12:]
        dividends_monthly = [{"month": m, "rate": round(r, 4)} for m, r in sorted_months]
        
        # DY 12m = sum of last 12 months dividends / current price
        total_12m = sum(r for _, r in sorted_months)
        dy_12m = round(total_12m / current_price, 4) if current_price and current_price > 0 else None
        
        # Last dividend = most recent rate
        last_dividend = cash_dividends[0].get("rate") if cash_dividends else None
        
        return dividends_monthly, dy_12m, last_dividend


    def _parse_date(date_str: str) -> datetime | None:
        """Parse date from BRAPI (handles YYYY-MM-DD and DD/MM/YYYY)."""
        if not date_str:
            return None
        for fmt in ("%Y-%m-%d", "%d/%m/%Y", "%Y-%m-%dT%H:%M:%S"):
            try:
                return datetime.strptime(date_str.split("T")[0] if "T" in date_str and fmt == "%Y-%m-%d" else date_str, fmt)
            except ValueError:
                continue
        return None
    ```

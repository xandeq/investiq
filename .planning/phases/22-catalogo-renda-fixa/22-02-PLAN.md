    const m = parseInt(minMonths, 10);
    if (!isNaN(m) && m > 0) {
      rows = rows.filter(r => r.min_months >= m);
    }
  }
  // Sort by net_pct for selectedPrazo (per D-05, D-06)
  rows = [...rows].sort((a, b) => {
    const aBd = a.ir_breakdowns.find(bd => bd.period_label === selectedPrazo);

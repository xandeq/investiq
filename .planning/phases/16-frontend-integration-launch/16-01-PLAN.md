```typescript
export function useAnalysisJob(jobId: string | null) {
  return useQuery({
    queryKey: ["ai", "job", jobId],
    queryFn: () => getJob(jobId!),
    enabled: !!jobId,
    refetchInterval: (query) => {
      const status = query.state.data?.status;
      if (status === "completed" || status === "failed") return false;
      return 2000;
    },
  });
}
```

From backend/app/modules/analysis/schemas.py (Backend response contracts):
```python
class AnalysisJobStatus(BaseModel):
    job_id: str
    status: str
    message: str | None = None

class AnalysisResponse(BaseModel):
    analysis_id: str
    analysis_type: str  # "dcf" | "earnings" | "dividend" | "sector"
    ticker: str
    status: str  # "pending" | "running" | "completed" | "failed" | "stale"
    result: dict | None = None
    data_metadata: DataMetadata | None = None
    disclaimer: str
    error_message: str | None = None

class DCFRequest(BaseModel):
    ticker: str  # min_length=4, max_length=10
    growth_rate: float | None = None  # 0-0.20
    discount_rate: float | None = None  # 0-0.30
    terminal_growth: float | None = None  # 0-0.05

class EarningsRequest(BaseModel):
    ticker: str

class DividendRequest(BaseModel):
    ticker: str

class SectorRequest(BaseModel):
    ticker: str
    max_peers: int = 10  # 3-15
```

Backend API endpoints (all return 202 with AnalysisJobStatus):
- POST /analysis/dcf (body: DCFRequest)
- POST /analysis/earnings (body: EarningsRequest)
- POST /analysis/dividend (body: DividendRequest)
- POST /analysis/sector (body: SectorRequest)
- GET /analysis/{job_id} (returns AnalysisResponse)

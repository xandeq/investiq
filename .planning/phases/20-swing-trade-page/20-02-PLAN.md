Create `frontend/src/features/swing_trade/types.ts`:
```typescript
export interface SwingSignalItem {
  ticker: string;
  name: string;
  sector: string;
  current_price: number;
  high_30d: number;
  discount_pct: number;
  dy: number | null;
  signal: "buy" | "sell" | "neutral";
  signal_strength: number;
  in_portfolio: boolean;
  quantity: number | null;
}

export interface SwingSignalsResponse {
  portfolio_signals: SwingSignalItem[];
  radar_signals: SwingSignalItem[];
  generated_at: string;
}

export interface SwingOperation {
  id: string;
  ticker: string;
  asset_class: string;
  quantity: number;
  entry_price: number;
  entry_date: string;
  target_price: number | null;
  stop_price: number | null;
  status: "open" | "closed" | "stopped";
  exit_price: number | null;
  exit_date: string | null;
  notes: string | null;
  current_price: number | null;
  pnl_pct: number | null;
  pnl_brl: number | null;
  days_open: number | null;
  target_progress_pct: number | null;
}

export interface OperationListResponse {
  open_count: number;
  closed_count: number;
  results: SwingOperation[];
}

export interface OperationCreatePayload {
  ticker: string;
  asset_class?: string;
  quantity: number;
  entry_price: number;
  entry_date: string;
  target_price?: number;
  stop_price?: number;
  notes?: string;
}
```

Create `frontend/src/features/swing_trade/api.ts`:
```typescript
import { apiClient } from "@/lib/api";
// fetchSignals: GET /swing-trade/signals → SwingSignalsResponse
// fetchOperations: GET /swing-trade/operations → OperationListResponse
// createOperation: POST /swing-trade/operations body=OperationCreatePayload
// closeOperation: PATCH /swing-trade/operations/{id}/close body={exit_price, exit_date?}
// deleteOperation: DELETE /swing-trade/operations/{id}
```
Same pattern as opportunity_detector/api.ts — use apiClient with credentials.
</action>

PART A — Create backend/tests/locustfile.py:

```python
"""Locust load test for InvestIQ Analysis API (Phase 15).

Tests 500 concurrent users hitting POST /analysis/dcf.

Usage (headless — run from project root):
  pip install locust
  locust -f backend/tests/locustfile.py \
    --host https://api.investiq.com.br \
    --users 500 \
    --spawn-rate 50 \
    --run-time 30s \
    --headless \
    --html backend/tests/locust-report.html

For local dev (requires running server):
  locust -f backend/tests/locustfile.py \
    --host http://localhost:8000 \
    --users 50 \
    --spawn-rate 10 \
    --run-time 30s \
    --headless \
    --html backend/tests/locust-report-local.html

Env vars required:
  LOAD_TEST_EMAIL    — email of a Pro/Enterprise test account
  LOAD_TEST_PASSWORD — password for the test account

IMPORTANT: Use a dedicated test tenant (not a production user account).
The load test will consume quota from this account.
"""
from __future__ import annotations

import os

from locust import HttpUser, between, task


class AnalysisUser(HttpUser):
    """Simulates a user requesting DCF analyses concurrently."""

    # Wait 20-100ms between tasks (aggressive — simulates burst traffic)
    wait_time = between(0.02, 0.1)

    def on_start(self) -> None:
        """Obtain a fresh JWT before making analysis requests.

        Never hardcode tokens — they expire. Always login via on_start().
        """
        email = os.environ.get("LOAD_TEST_EMAIL", "loadtest@investiq.com.br")
        password = os.environ.get("LOAD_TEST_PASSWORD", "")

        if not password:
            raise ValueError(
                "LOAD_TEST_PASSWORD env var is required for load test. "
                "Set it to the password of a Pro/Enterprise test account."
            )

        response = self.client.post(
            "/auth/login",
            json={"email": email, "password": password},
        )
        if response.status_code == 200:
            token = response.json().get("access_token", "")
            self.headers = {"Authorization": f"Bearer {token}"}
        else:
            # Login failed — mark this user as broken so errors show in report
            self.headers = {}
            self.environment.runner.quit()

    @task
    def request_dcf(self) -> None:
        """Request a DCF analysis for PETR4.

        Expects 202 (queued) or 429 (rate limited) or 403 (quota exceeded).
        All are valid responses under load — only 500 is a test failure.
        """
        response = self.client.post(
            "/analysis/dcf",
            json={"ticker": "PETR4"},
            headers=self.headers,
            # Mark 429 and 403 as expected (not failures in Locust report)
            catch_response=True,
        )
        with response:
            if response.status_code in (202, 429, 403):
                response.success()
            elif response.status_code >= 500:
                response.failure(
                    f"Server error: {response.status_code} — {response.text[:200]}"
                )
            else:
                response.failure(f"Unexpected status: {response.status_code}")

    @task(weight=2)
    def poll_job_status(self) -> None:
        """Poll a known job ID to measure completion latency.

        Uses a placeholder job ID — mainly measures routing overhead.
        Real p95 completion latency must be measured by polling actual job IDs.
        This task is lower priority than request_dcf (weight=2 vs 5 default).
        """
        # Use a known-invalid UUID to test route handling without real jobs
        self.client.get(
            "/analysis/00000000-0000-0000-0000-000000000000",
            headers=self.headers,
            catch_response=True,
        )
```

```python
"""Performance smoke tests for Phase 15 (pytest, not Locust).

These run in CI. They test API concurrency behavior with a small number
of requests (10) to avoid load testing in CI environments.

Full 500-user load test runs via Locust headlessly (see locustfile.py).
"""
from __future__ import annotations

import threading
import time
from unittest.mock import MagicMock, patch

import pytest


class TestConcurrentJobSubmission:
    """Smoke test: 10 concurrent DCF requests must all return 202 or valid error."""

    def test_10_concurrent_dcf_submissions_all_return_202(self, test_client, auth_headers):
        """10 simultaneous POST /analysis/dcf requests must not produce any 500 errors."""
        results = []
        errors = []

        def submit_request():
            try:
                resp = test_client.post(
                    "/analysis/dcf",
                    json={"ticker": "PETR4"},
                    headers=auth_headers,
                )
                results.append(resp.status_code)
            except Exception as exc:
                errors.append(str(exc))

        threads = [threading.Thread(target=submit_request) for _ in range(10)]
        for t in threads:
            t.start()
        for t in threads:
            t.join(timeout=30)

        assert not errors, f"Thread errors: {errors}"
        # All responses must be one of: 202 (queued), 429 (rate limited), 403 (quota exceeded)
        # None should be 500 (server error) or connection failure
        for status_code in results:
            assert status_code in (202, 429, 403), (
                f"Unexpected status {status_code} — server error or misconfiguration"
            )

    def test_job_ids_are_unique(self, test_client, auth_headers):
        """Concurrent requests that succeed (202) must return distinct job_ids."""
        job_ids = []
        lock = threading.Lock()

        def submit_and_collect():
            resp = test_client.post(
                "/analysis/dcf",
                json={"ticker": "VALE3"},
                headers=auth_headers,
            )
            if resp.status_code == 202:
                with lock:
                    job_ids.append(resp.json().get("job_id"))

        threads = [threading.Thread(target=submit_and_collect) for _ in range(5)]
        for t in threads:
            t.start()
        for t in threads:
            t.join(timeout=30)

        if len(job_ids) > 1:
            assert len(set(job_ids)) == len(job_ids), "Duplicate job_ids detected"

    def test_quota_exhaustion_returns_403_not_500(self, test_client, auth_headers_no_quota):
        """When quota is exhausted, server returns 403 with QUOTA_EXCEEDED code (not 500)."""
        resp = test_client.post(
            "/analysis/dcf",
            json={"ticker": "PETR4"},
            headers=auth_headers_no_quota,
        )
        assert resp.status_code == 403
        body = resp.json()
        assert body.get("detail", {}).get("code") == "QUOTA_EXCEEDED"
```

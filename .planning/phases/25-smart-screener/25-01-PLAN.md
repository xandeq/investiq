Then add service function:
```python
async def get_complementary_assets(
    tenant_db: AsyncSession,
    global_db: AsyncSession,
    tenant_id: str,
    limit: int = 100,
) -> list[ComplementaryAssetRow]:
    """
    Return screener universe filtered to assets NOT in user's portfolio sectors.
    Ranked by relevance to identified portfolio gaps.
    """
    
    # 1. Get portfolio health to identify current sectors
    health = await compute_portfolio_health(tenant_db, global_db, tenant_id)
    
    # 2. If no portfolio, return full universe (all assets are complementary)
    if not health.has_portfolio:
        result = await global_db.execute(
            select(ScreenerSnapshot)
            .where(
                ScreenerSnapshot.snapshot_date == (
                    select(func.max(ScreenerSnapshot.snapshot_date))
                )
            )
            .limit(limit)
        )
        snaps = result.scalars().all()
        return [
            ComplementaryAssetRow(
                ticker=s.ticker,
                sector=s.sector,
                preco_atual=s.preco_atual,
                dy_12m_pct=s.dy_12m_pct,
                variacao_12m_pct=s.variacao_12m_pct,
                market_cap=s.market_cap,
                relevance_score=50.0,  # Neutral score for empty portfolio
            )
            for s in snaps
        ]
    
    # 3. Get sectors from portfolio (from compute_portfolio_health)
    # sector_map is computed inside compute_portfolio_health
    # We need to extract it — for now, recompute sector analysis
    tx_result = await tenant_db.execute(
        select(Transaction.ticker).where(
            Transaction.tenant_id == tenant_id,
            Transaction.transaction_type.in_(["buy", "sell"]),
            Transaction.deleted_at.is_(None),
        )
    )
    portfolio_tickers = {row[0] for row in tx_result.all()}
    
    # Get sectors for portfolio tickers
    portfolio_sectors = set()
    if portfolio_tickers:
        snap_result = await global_db.execute(
            select(ScreenerSnapshot.sector).where(
                ScreenerSnapshot.ticker.in_(list(portfolio_tickers)),
                ScreenerSnapshot.snapshot_date == (
                    select(func.max(ScreenerSnapshot.snapshot_date))
                ),
            )
        )
        portfolio_sectors = {row[0] for row in snap_result.all() if row[0]}
    
    # 4. Get latest screener snapshot date
    date_result = await global_db.execute(
        select(func.max(ScreenerSnapshot.snapshot_date))
    )
    latest_date = date_result.scalar()
    
    # 5. Query complementary assets (NOT in portfolio sectors)
    query = select(ScreenerSnapshot).where(
        ScreenerSnapshot.snapshot_date == latest_date,
        (ScreenerSnapshot.sector.is_(None)) | (ScreenerSnapshot.sector.notin_(portfolio_sectors)),
    )
    
    result = await global_db.execute(query)
    snaps = result.scalars().all()
    
    # 6. Score by relevance
    rows = []
    for s in snaps:
        # Relevance = DY (prefer income) + inverse of concentration (prefer underrepresented sectors)
        score = float(s.dy_12m_pct or 0) * 2 + (50 - float(s.variacao_12m_pct or 50))
        rows.append(
            ComplementaryAssetRow(
                ticker=s.ticker,
                sector=s.sector,
                preco_atual=s.preco_atual,
                dy_12m_pct=s.dy_12m_pct,
                variacao_12m_pct=s.variacao_12m_pct,
                market_cap=s.market_cap,
                relevance_score=min(100, max(0, score)),
            )
        )
    
    # 7. Sort by relevance, cap at limit
    rows.sort(key=lambda x: x.relevance_score, reverse=True)
    return rows[:limit]
```

```typescript
import { useQuery } from "@tanstack/react-query";

export interface ComplementaryAsset {
  ticker: string;
  sector: string | null;
  preco_atual: number | null;
  dy_12m_pct: number | null;
  variacao_12m_pct: number | null;
  market_cap: number | null;
  relevance_score: number;
}

export function useSmartScreener(enabled?: boolean) {
  return useQuery({
    queryKey: ["smart-screener"],
    queryFn: async () => {
      const res = await fetch("/api/advisor/screener");
      if (!res.ok) throw new Error(`Failed to fetch: ${res.status}`);
      return res.json() as Promise<ComplementaryAsset[]>;
    },
    enabled: enabled !== false,
    staleTime: 10 * 60 * 1000, // 10 minutes
  });
}
```

```typescript
interface SmartScreenerSectionProps {
  assets: ComplementaryAsset[];
  isLoading: boolean;
}

function SmartScreenerSection({ assets, isLoading }: SmartScreenerSectionProps) {
  const [filterSector, setFilterSector] = React.useState<string>("");
  
  const filtered = React.useMemo(() => {
    return assets.filter(a => !filterSector || a.sector === filterSector);
  }, [assets, filterSector]);
  
  const sectors = React.useMemo(() => {
    return Array.from(new Set(assets.map(a => a.sector).filter(Boolean)));
  }, [assets]);
  
  if (isLoading) return <div>Carregando Smart Screener...</div>;
  if (!assets.length) return <div>Nenhum ativo complementar encontrado</div>;
  
  return (
    <div className="border rounded-lg p-6 mb-8">
      <h2 className="text-lg font-semibold mb-4">Smart Screener</h2>
      
      <div className="mb-4">
        <select
          value={filterSector}
          onChange={(e) => setFilterSector(e.target.value)}
          className="px-3 py-2 border rounded"
        >
          <option value="">Todos os setores ({filtered.length})</option>
          {sectors.map(s => (
            <option key={s} value={s || ""}>{s} ({assets.filter(a => a.sector === s).length})</option>
          ))}
        </select>
      </div>
      
      <table className="w-full text-sm">
        <thead className="border-b">
          <tr>
            <th className="text-left px-2 py-2">Ticker</th>
            <th className="text-left px-2 py-2">Setor</th>
            <th className="text-right px-2 py-2">DY 12m</th>
            <th className="text-right px-2 py-2">Var. 12m</th>
            <th className="text-right px-2 py-2">Preço</th>
          </tr>
        </thead>
        <tbody>
          {filtered.slice(0, 50).map(a => (
            <tr key={a.ticker} className="border-b hover:bg-gray-50">
              <td className="px-2 py-2">
                <a href={`/stock/${a.ticker}`} className="text-blue-600 hover:underline">
                  {a.ticker}
                </a>
              </td>
              <td className="px-2 py-2 text-xs">{a.sector || "—"}</td>
              <td className="text-right px-2 py-2 text-xs">{a.dy_12m_pct?.toFixed(2)}%</td>
              <td className={`text-right px-2 py-2 text-xs ${(a.variacao_12m_pct || 0) > 0 ? "text-green-600" : "text-red-600"}`}>
                {a.variacao_12m_pct?.toFixed(2)}%
              </td>
              <td className="text-right px-2 py-2">R$ {a.preco_atual?.toFixed(2)}</td>
            </tr>
          ))}
        </tbody>
      </table>
      
      {filtered.length > 50 && (
        <p className="text-xs text-muted-foreground mt-2">
          Mostrando 50 de {filtered.length} resultados
        </p>
      )}
    </div>
  );
}
```

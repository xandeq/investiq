    ```python
    def compute_import_hash(tenant_id: str, ticker: str, txn_type: str,
                             txn_date, quantity: Decimal, unit_price: Decimal) -> str:
        import hashlib
        parts = "|".join([tenant_id, ticker.upper(), txn_type.lower(), str(txn_date),
                          f"{quantity:.8f}", f"{unit_price:.8f}"])
        return hashlib.sha256(parts.encode("utf-8")).hexdigest()
    ```

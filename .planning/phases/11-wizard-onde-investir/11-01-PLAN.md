```python
from app.modules.wizard.models import WizardJob

@pytest.mark.asyncio
async def test_get_wizard_job_returns_delta_on_completion(client, email_stub, db_session):
    user_id = await register_verify_and_login(client, email_stub)
    # Insert a completed wizard job directly into DB
    completed_job = WizardJob(
        tenant_id=user_id,
        perfil="moderado",
        prazo="1a",
        valor=10000.0,
        status="completed",
        result={
            "allocation": {"acoes_pct": 40, "fiis_pct": 20, "renda_fixa_pct": 30, "caixa_pct": 10, "rationale": "Texto generico"},
            "delta": [{"asset_class": "acoes", "label": "Acoes", "current_pct": 20.0, "suggested_pct": 40.0, "delta_pct": 20.0, "action": "add", "valor_delta": 2000.0}]
        }
    )
    db_session.add(completed_job)
    await db_session.commit()
    await db_session.refresh(completed_job)

    resp = await client.get(f"/wizard/{completed_job.id}")
    assert resp.status_code == 200
    data = resp.json()
    assert data["status"] == "completed"
    assert "result" in data
    assert "delta" in data["result"]
    assert len(data["result"]["delta"]) == 1
    assert data["result"]["delta"][0]["asset_class"] == "acoes"
    assert data["result"]["delta"][0]["suggested_pct"] == 40.0
    assert data["result"]["delta"][0]["action"] == "add"
```

async def start_portfolio_advisor_job(
    tenant_db: AsyncSession,
    global_db: AsyncSession,
    tenant_id: str,
    user_id: str,
) -> str:
    """
    Start async portfolio advisor job. Returns job_id immediately.
    Job runs in background (via Celery or sync processing).
    """
    from uuid import uuid4
    from datetime import datetime
    from app.modules.wizard.models import WizardJob  # Reuse existing async job model
    
    job_id = str(uuid4())
    
    # Create job record in DB
    job = WizardJob(
        id=job_id,
        tenant_id=tenant_id,
        user_id=user_id,
        job_type="portfolio_advisor",
        status="pending",
        created_at=datetime.utcnow(),
    )
    tenant_db.add(job)
    await tenant_db.commit()
    
    # Queue background task (sync processing for now — can move to Celery later)
    # For MVP: start processing immediately, update job status in background thread
    import asyncio
    asyncio.create_task(_process_portfolio_advisor_job(job_id, tenant_id, user_id, tenant_db, global_db))
    
    return job_id


async def _process_portfolio_advisor_job(
    job_id: str,
    tenant_id: str,
    user_id: str,
    tenant_db: AsyncSession,
    global_db: AsyncSession,
) -> None:
    """Background task to run portfolio advisor and store result."""
    from app.modules.wizard.models import WizardJob
    from datetime import datetime
    
    try:
        # Mark as running
        job = await tenant_db.get(WizardJob, job_id)
        if job:
            job.status = "running"
            await tenant_db.commit()
        
        # Fetch portfolio health (reuse compute_portfolio_health for context)
        health = await compute_portfolio_health(tenant_db, global_db, tenant_id)
        
        # Fetch portfolio positions for AI input
        positions = await get_portfolio_positions(tenant_db, tenant_id)
        
        # Fetch macro data context
        macro_data = await get_macro_context(global_db)
        
        # Prepare input for AI skill
        positions_list = [
            {
                "ticker": p.ticker,
                "quantity": float(p.quantity),
                "cost_basis": float(p.cost_basis),
                "current_price": float(p.current_price),
            }
            for p in positions
        ]
        
        p_l = {
            "total": float(health.passive_income_monthly_brl * 12),  # Approximate
        }
        
        allocation = {
            "biggest_sector_pct": float(health.biggest_risk.split("%")[0]) if health.biggest_risk else 0,
            "concentration": float(health.health_score) / 100,
        }
        
        # Call AI skill
        result_dict = await run_portfolio_advisor(
            positions=positions_list,
            p_l=p_l,
            allocation=allocation,
            macro_data=macro_data,
        )
        
        # Store result
        job = await tenant_db.get(WizardJob, job_id)
        if job:
            job.status = "completed"
            job.result = {
                "diagnostico": result_dict.get("diagnostico", ""),
                "pontos_positivos": result_dict.get("pontos_positivos", []),
                "pontos_de_atencao": result_dict.get("pontos_de_atencao", []),
                "sugestoes": result_dict.get("sugestoes", []),
                "proximos_passos": result_dict.get("proximos_passos", []),
                "health_score": health.health_score,
                "biggest_risk": health.biggest_risk,
                "passive_income_monthly_brl": str(health.passive_income_monthly_brl),
                "underperformers": health.underperformers,
                "provider_used": "claude-api",
            }
            job.completed_at = datetime.utcnow()
            await tenant_db.commit()
    
    except Exception as exc:
        logger.error(f"Portfolio advisor job {job_id} failed: {exc}")
        job = await tenant_db.get(WizardJob, job_id)
        if job:
            job.status = "failed"
            job.error_message = str(exc)
            await tenant_db.commit()


async def get_advisor_job_result(
    tenant_db: AsyncSession,
    tenant_id: str,
    job_id: str,
) -> dict:
    """Fetch job status and result from DB."""
    from app.modules.wizard.models import WizardJob
    
    job = await tenant_db.get(WizardJob, job_id)
    if not job or job.tenant_id != tenant_id:
        return None
    
    return {
        "job_id": job.id,
        "status": job.status,
        "result": job.result,
        "error_message": job.error_message,
        "created_at": job.created_at,
        "completed_at": job.completed_at,
    }

from app.modules.advisor.service import start_portfolio_advisor_job, get_advisor_job_result
from app.modules.advisor.schemas import AdvisorStartResponse, AdvisorJobResponse, AdvisorResult
from app.modules.billing.service import check_subscription_tier

@router.post(
    "/analyze",
    response_model=AdvisorStartResponse,
    summary="Start async portfolio advisor job",
    tags=["advisor"],
    status_code=202,  # Accepted
)
@limiter.limit("5/minute")
async def start_advisor_analysis(
    request: Request,
    current_user: dict = Depends(get_current_user),
    tenant_db: AsyncSession = Depends(get_tenant_db),
    global_db: AsyncSession = Depends(get_global_db),
) -> AdvisorStartResponse:
    """Start async AI portfolio analysis job. Returns immediately with job_id for polling."""
    
    # Check premium tier
    if not await check_subscription_tier(current_user, "premium"):
        raise HTTPException(
            status_code=403,
            detail="Portfolio advisor requires premium tier"
        )
    
    tenant_id = current_user["tenant_id"]
    user_id = current_user["user_id"]
    
    job_id = await start_portfolio_advisor_job(tenant_db, global_db, tenant_id, user_id)
    
    return AdvisorStartResponse(
        job_id=job_id,
        status="pending",
    )


@router.get(
    "/analyze/{job_id}",
    response_model=AdvisorJobResponse,
    summary="Poll advisor job result",
    tags=["advisor"],
)
@limiter.limit("30/minute")
async def get_advisor_analysis(
    job_id: str,
    request: Request,
    current_user: dict = Depends(get_current_user),
    tenant_db: AsyncSession = Depends(get_tenant_db),
) -> AdvisorJobResponse:
    """Poll job status. Returns result when completed."""
    
    tenant_id = current_user["tenant_id"]
    job_data = await get_advisor_job_result(tenant_db, tenant_id, job_id)
    
    if not job_data:
        raise HTTPException(status_code=404, detail="Job not found")
    
    return AdvisorJobResponse(
        job_id=job_data["job_id"],
        status=job_data["status"],
        result=AdvisorResult(**job_data["result"]) if job_data["result"] else None,
        error_message=job_data["error_message"],
        created_at=job_data["created_at"],
        completed_at=job_data["completed_at"],
    )
